123  /**
124  * Computes the number of lines a MultiCell of width w will take

126  * @param $txt The text
127  */
128  public function NbLines($w, $txt) {
129  $cw=&$this->CurrentFont['cw'];
130  if($w==0) {
131  $w=$this->w-$this->rMargin-$this->x;
132  }
133  $wmax=($w-2*$this->cMargin)*1000/$this->FontSize;
134  $s=str_replace("\r", '', $txt);
135  $nb=strlen($s);
136  if($nb>0 && $s[$nb-1]=="\n") {
137  $nb--;
138  }
139  $sep=-1;
140  $i=0;
141  $j=0;
142  $l=0;
143  $nl=1;
144  while($i<$nb) {
145  $c=$s[$i];
146  if($c=="\n") {
147  $i++;
148  $sep=-1;
149  $j=$i;
150  $l=0;
151  $nl++;
152  continue;
153  }
154  if($c==' ') {
155  $sep=$i;
156  }
157  $l+=$cw[$c];
158  if($l>$wmax) {
159  if($sep==-1) {
160  if($i==$j) {
161  $i++;
162  }
163  }
164  else {
165  $i=$sep+1;
166  }
167  $sep=-1;
168  $j=$i;
169  $l=0;
170  $nl++;
171  }
172  else {
173  $i++;
174  }
175  }
176  return $nl;
177  }
