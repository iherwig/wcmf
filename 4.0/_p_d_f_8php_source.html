121  /**
122  * Computes the number of lines a MultiCell of width w will take
123  * @param $w The width
124  * @param $txt The text
125  */
126  public function NbLines($w, $txt) {
127  $cw=&$this->CurrentFont['cw'];
128  if($w==0) {
129  $w=$this->w-$this->rMargin-$this->x;
130  }
131  $wmax=($w-2*$this->cMargin)*1000/$this->FontSize;
132  $s=str_replace("\r",'',$txt);
133  $nb=strlen($s);
134  if($nb>0 && $s[$nb-1]=="\n") {
135  $nb--;
136  }
137  $sep=-1;
138  $i=0;
139  $j=0;
140  $l=0;
141  $nl=1;
142  while($i<$nb) {
143  $c=$s[$i];
144  if($c=="\n") {
145  $i++;
146  $sep=-1;
147  $j=$i;
148  $l=0;
149  $nl++;
150  continue;
151  }
152  if($c==' ') {
153  $sep=$i;
154  }
155  $l+=$cw[$c];
156  if($l>$wmax) {
157  if($sep==-1) {
158  if($i==$j) {
159  $i++;
160  }
161  }
162  else {
163  $i=$sep+1;
164  }
165  $sep=-1;
166  $j=$i;
167  $l=0;
168  $nl++;
169  }
170  else {
171  $i++;
172  }
173  }
174  return $nl;
175  }
